    def initialize
      @path = "#{Rails.root}/log/user_agents.log"
      @ua_logger = ActiveSupport::BufferedLogger.new(@path)
      add_first_break if File.read(@path).scan(/\n/).empty?
      silence_warnings {UserAgentLog.const_set "UA_LOGGER", @ua_logger}
    end
