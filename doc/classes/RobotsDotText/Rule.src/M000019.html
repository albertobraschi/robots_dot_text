    def to_s
      string = String.new
      if @user_agent.is_a?(Array)
        @user_agent.each do |ua|
          string << "User-agent: #{known_agents ua}\n"
        end
      else
        string << "User-agent: #{known_agents @user_agent}\n"
      end
      @paths.each { |path| string << "Disallow: #{path}\n" }
      string
    end
